% Design system: local/source=orange!20, Kontablo/universal=blue!15,

%   Arrows >=Latex, thick. rounded corners=4pt.
\begin{figure}[ht]
\centering
\resizebox{0.65\textwidth}{!}{%
\begin{tikzpicture}[
    >=Latex,
    node distance=1.8cm,
    procnode/.style={draw=blue!60!black, rectangle, rounded corners=4pt,
        fill=blue!10, minimum width=4.2cm, minimum height=1.0cm,
        align=center, font=\small},
    checknode/.style={draw=yellow!70!black, rectangle, rounded corners=4pt,
        fill=yellow!20, minimum width=4.2cm, minimum height=1.0cm,
        align=center, font=\small},
    approvednode/.style={draw=green!60!black, rectangle, rounded corners=4pt,
        fill=green!12, minimum width=3.0cm, minimum height=1.0cm,
        align=center, font=\small},
    flagnode/.style={draw=red!60!black, rectangle, rounded corners=4pt,
        fill=red!10, minimum width=3.0cm, minimum height=1.0cm,
        align=center, font=\small},
    edgelabel/.style={font=\scriptsize, fill=white, inner sep=1pt}
]
    \node[procnode]  (step1) {AI Proposed Mapping};
    \node[checknode, below=1.8cm of step1] (step2) {Debit/Credit Check};
    \node[checknode, below=1.8cm of step2] (step3) {Liquidity Boundary};
    \node[approvednode, right=3.5cm of step3] (success) {Approved};
    \node[flagnode,   left=3.5cm of step3]  (fail)    {Raise Audit Flag};

    \draw[->, thick] (step1) -- (step2);
    \draw[->, thick] (step2) -- node[edgelabel, right] {Pass} (step3);
    \draw[->, thick] (step3) -- node[edgelabel, above] {Pass} (success);
    \draw[->, thick] (step2.west) -- ++(-1.2,0) |-
        node[edgelabel, near start, left] {Fail} (fail.north);
    \draw[->, thick] (step3) -- node[edgelabel, above] {Fail} (fail);
\end{tikzpicture}%
}
\caption{The deterministic boundary checklist verifies AI-proposed mappings
prior to ledger write-in.}
