% Figure: Historical evolution of financial standards
% Source: originally in sections/precedents.tex
% Depicts: Timeline 1937 → present; reporting and payment standardization vs.
%          persistent execution-layer fragmentation, resolved by Kontablo.
%
% Design system: local/source=orange!20, Kontablo/universal=blue!15,
%   IFRS/output=green!15, decision=yellow!25, database=gray!20,
%   agent=violet!15. Arrows >=Latex, thick. rounded corners=4pt.
\begin{figure}[ht]
\centering
\resizebox{\textwidth}{!}{%
\begin{tikzpicture}[
    >=Latex,
    node distance=1.5cm and 2.2cm,
    milestone/.style={circle, fill=blue!60!black, draw=blue!60!black, inner sep=2.5pt},
    descnode/.style={draw=gray!60, rectangle, rounded corners=4pt,
        fill=gray!10, text width=4.2cm, align=left, font=\scriptsize,
        minimum height=2.2cm}
]

% Timeline axis
\draw[thick, ->, gray!60] (0,0) -- (15.5,0)
    node[right, font=\small\bfseries, black] {Present};

% 1937/1947
\node[milestone] (m1) at (1,0) {};
\node[above=0.25cm of m1, font=\bfseries\small] {1937/1947};
\node[descnode, fill=orange!12, draw=orange!50, below=0.4cm of m1] (d1) {%
    \textbf{National Charts-of-Accounts}\\[2pt]%
    German \textit{Gemeinschaftsrahmen} (1937), French \textit{Plan Comptable}
    (1947). Localized, rigid ledger trees.};

% Late 1990s
\node[milestone] (m2) at (4.5,0) {};
\node[above=0.25cm of m2, font=\bfseries\small] {Late 1990s};
\node[descnode, fill=orange!12, draw=orange!50, below=0.4cm of m2] (d2) {%
    \textbf{Electronic Invoicing (EDI)}\\[2pt]%
    UN/EDIFACT standardizes invoice transport but does not specify
    ledger-account mapping rules.};

% 2001/2004
\node[milestone] (m3) at (8,0) {};
\node[above=0.25cm of m3, font=\bfseries\small] {2001/2004};
\node[descnode, fill=orange!12, draw=orange!50, below=0.4cm of m3] (d3) {%
    \textbf{Reporting Focus (XBRL)}\\[2pt]%
    Regulator-focused financial statement tagging. Excludes
    transaction-level mapping or execution semantics.};

% 2004/2013
\node[milestone] (m4) at (11.5,0) {};
\node[above=0.25cm of m4, font=\bfseries\small] {2004/2013};
\node[descnode, fill=orange!12, draw=orange!50, below=0.4cm of m4] (d4) {%
    \textbf{Payment Messaging (ISO~20022)}\\[2pt]%
    Standardizes bank-to-bank messaging but leaves ERP general-ledger
    posting and statutory mapping unaddressed.};

% Kontablo (present)
\node[milestone, fill=blue!60!black, draw=blue!60!black] (m5) at (14.5,0) {};
\node[above=0.25cm of m5, font=\bfseries\small] {2025};
\node[descnode, fill=blue!10, draw=blue!50, below=0.4cm of m5] (d5) {%
    \textbf{Graph-Based Subledger (Kontablo)}\\[2pt]%
    Unifies reporting and execution layers for ERPs and autonomous
    AI agents in a single graph-ontology API.};

\end{tikzpicture}%
}
\caption{Historical evolution of financial standards. While communication
(payment messaging) and disclosure (XBRL) standardized over time, the
execution layer (general ledger chart-of-accounts mappings) remained
fragmented at the national tree level until the introduction of Kontablo's
graph ontology.}
\label{fig:historical_timeline}
\end{figure}
