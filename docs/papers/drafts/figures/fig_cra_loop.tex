% Figure: Co-responsibility Architecture (CRA) feedback loop
% Source: originally in sections/governance.tex
% Depicts: AI Proposes → Human Disposes → Rule Engine validates →
%          Inconsistency Flagged + Audit Ledger (immutable).
\begin{figure}[ht]
\centering
\resizebox{\textwidth}{!}{%
\begin{tikzpicture}[
    >=Latex,
    node distance=2cm,
    block/.style={draw, rectangle, rounded corners, fill=blue!5, text width=3.5cm, align=center, minimum height=1.2cm, font=\small},
    decision/.style={draw, diamond, aspect=2, fill=yellow!10, text width=3cm, align=center, font=\small},
    database/.style={draw, cylinder, shape border rotate=90, fill=gray!15, text width=2.5cm, align=center, minimum height=1.5cm, font=\small}
]

% Nodes
\node[block] (ai_propose) {\textbf{1. AI Proposes}\\ Generates mapping proposal and confidence score};
\node[block, right=1.5cm of ai_propose] (human_override) {\textbf{2. Human Disposes}\\ Accountant reviews or overrides mapping};
\node[decision, below=1cm of human_override] (rule_check) {\textbf{3. Rule Engine}\\ Validates deterministic boundaries};

\node[block, left=2.3cm of rule_check] (flag_record) {\textbf{4. Inconsistency Flagged}\\ Set \texttt{inconsistency\_flag} = True \& write audit note};
\node[database, below=1cm of flag_record] (db) {\textbf{Audit Ledger}\\ Persist signed transactional metadata};

% Paths
\draw[->, thick] (ai_propose) -- (human_override);
\draw[->, thick] (human_override) -- (rule_check);
\draw[->, thick] (rule_check) -- node[above] {Violation} (flag_record);
\draw[->, thick] (rule_check) -- node[right] {Valid} (db);
\draw[->, thick] (flag_record) -- (db);

\end{tikzpicture}%
}
\caption{The Co-responsibility Architecture (CRA) feedback loop. An AI-proposed mapping is reviewed by a human accountant. If the human override violates deterministic ontology rules (e.g., changing cash nature), the system forces the entry of an explanation, flags the transaction, and writes it to the immutable ledger for future auditor review.}
\label{fig:cra_loop}
\end{figure}
